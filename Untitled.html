import pandas as pd
import matplotlib.pyplot as plt

df = pd.read_csv('Titanic-Dataset.csv')
print("Task 3 - EDA by Nikhil | Shape:", df.shape)

plt.figure(figsize=(12,8))

plt.subplot(2,2,1)
df['Survived'].value_counts().plot(kind='bar')
plt.title('1. Survival Count - 0=Died, 1=Survived')

plt.subplot(2,2,2)
df['Age'].dropna().hist(bins=30)
plt.title('2. Age Distribution')

plt.subplot(2,2,3)
df.groupby('Sex')['Survived'].mean().plot(kind='bar')
plt.title('3. Female vs Male Survival Rate')

plt.subplot(2,2,4)
df.corr(numeric_only=True)['Survived'].sort_values().plot(kind='barh')
plt.title('4. Correlation with Survival')

plt.tight_layout()
plt.show()

print("\n--- KEY FINDINGS ---")
print("1. Females ku 74% survival rate - males kanna 4x ekkuva")
print("2. 1st Class passengers ekkuva survive ayyaru") 
print("3. Age column lo 177 missing values unnay")
